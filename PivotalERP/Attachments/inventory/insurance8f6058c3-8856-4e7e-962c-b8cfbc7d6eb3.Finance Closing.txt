with tbl_FLed
as
(
select T.LedgerId,Sum(T.DrAmount) As DrAmount,Sum(T.CrAmount) as CrAmount from @tmpTran T group by T.LedgerId
)
select Led.LedgerId,Led.LedgerName,LG.LedgerGroupId,LG.GroupName,(case when Lo.Amt is null then 0 else Lo.Amt end) as Amt,(case when FL.DrAmount is null then 0 else FL.DrAmount end) as DrAmount,(case when FL.CrAmount is null then 0 else FL.CrAmount end) as CrAmount,Led.LedgerGroupId as LedgerGroupId1 into tmpLedgerClosing from TBL_LEDGER Led left outer join tbl_FLed FL on FL.LedgerId=Led.LedgerId left outer join f_GetLedgerOpeningBalance(@DateFrom,0) LO on Led.LedgerId=Lo.Id 
inner join TBL_LEDGERGROUP LG on LG.LedgerGroupId=Led.LedgerGroupId
order by Led.LedgerName;


GO

declare @LedgerId int;
declare @Amount float;
declare cur cursor for select LedgerId,Closing from tmpLedgerClosing
open cur
fetch next from cur into @LedgerId,@Amount
while @@Fetch_Status=0
begin 
if @Amount>0 
update tbl_ledger set DrCr=1,Amount=@Amount where LedgerId=@LedgerId;
else
update tbl_ledger set DrCr=2,Amount=abs(@Amount) where LedgerId=@LedgerId;
fetch next from cur into @LedgerId,@Amount
end
close cur
deallocate cur
